const genres = ["classic", "pop", "classic", "classic", "pop"];
const plays = [500, 600, 150, 800, 2500];
function solution(genres, plays) {
    var answer = [];
    // 장르와 재생횟수로 객체를 만든다
    const songsInfo= genres.map((one, index)=>{
        return {
            id: index,
            genre: one,
            play: plays[index]
        }
    });
    // 각 장르의 재생횟수 저장용 배열
    let playCountForEachGenre = [];
    songsInfo.forEach(song => {
        // 장르의 재생횟수를 있으면 더하고 없으면 생성한다
        if(!playCountForEachGenre.find(genrePlay => genrePlay.genre === song.genre)){
            playCountForEachGenre.push({
                genre: song.genre,
                play: song.play
            });
        }else{
            playCountForEachGenre.find(genrePlay => genrePlay.genre === song.genre).play += song.play;
        }
        // 장르의 재생횟수로 정렬한다
        playCountForEachGenre.sort((a,b)=> b.play-a.play);
    });
    // 장르별로 분류를 하고, 분류된 배열을 sorting하고, 하나만 있으면 넣고 끝, 하나 더 있으면 그거 넣고 끝
    playCountForEachGenre.forEach(genrePlay => {
        let thisGenreSongs = songsInfo.filter(song => song.genre === genrePlay.genre);
        thisGenreSongs.sort((a,b)=>b.play-a.play);
        answer.push(thisGenreSongs[0].id);
        if(thisGenreSongs.length >1){
            answer.push(thisGenreSongs[1].id);
        }
    });
    return answer;
}
console.log(solution(genres, plays));

function solution(progresses, speeds) {
    var answer = [0];
    let currentDay=0;
    // 각 작업의 남은 일수를 가지고 있는 배열을 만든다.
    const leftDays = (progresses.map((one, index)=>Math.ceil((100-one)/speeds[index])));
    // 처음 max 값은 첫 작업의 남은 일수로 설정한다.
    let maxDay = leftDays[0];

    // i와 j로 루프를 돌면서 j를 이용해서 answer 배열의 다음방으로 넘어갈 수 있게 설정
    for(let i = 0, j = 0; i< leftDays.length; i++){
        // max가 크다면 루프를 돌면서 작업의 갯수를 더하고, 맥스보다 큰 값이 나오면, 그 다음에 서비스에 반영 되는 작업이니, 새로운 방으로 가서 1을 대입한다
        if( maxDay >= leftDays[i] ) {
            answer[j] += 1;
        } else {
            maxDay = leftDays[i];
            answer[++j] = 1;
        }
    }

    return answer;}

const jobs = [[0, 3], [1, 9], [2, 6]];
function solution(jobs) {
    var answer = 0;
    const numberOfJobs = jobs.length;

    //일단 요청 된 순서대로 정렬이 필요
    const sortedJobs = jobs.sort((a,b)=>a[0] === b[0] ? a[1] - b[1] : a[0] -b[0]);

    //시간은 정렬된 일들의 첫번째 요청시간
    let currentTime = sortedJobs[0][0];
    let que = [];

    while(sortedJobs.length > 0) {
        let currentJob = sortedJobs[0];
        // 처음에는 당연히 여기 안들어온다
        if(currentJob[0] < currentTime) {
            que.push(currentJob)
            sortedJobs.shift();
        } else {
            if(que.length > 0 ) {
                que.sort((a,b)=>a[1]-b[1]);
                let waitJob = que.shift();
                currentTime += waitJob[1];
                //소요시간을 answer에 축적
                answer += currentTime - waitJob[0];

            } else {
                //맨처음 들어올 곳 여기서 처음 걸리는 시간이 answer에 더해지고, shift되면서 다음 while loop로 들어간다
                currentTime = currentJob[0] + currentJob[1];
                answer += currentJob[1];
                sortedJobs.shift();
            }
        }
        // if there are are tasks only in que
        if(sortedJobs.length === 0 && que.length > 0) {
            que.sort((a,b)=>a[1]-b[1]);
            while(que.length > 0) {
                let waitJob = que.shift();
                //현재 걸린시간에 누적을 해야하니 더해준다
                currentTime += waitJob[1];
                //answer는 걸린시간만 더하면 되니깐 현재시간에서 요청시간을 빼서 처리시간만 answer에 더해준다
                answer += currentTime - waitJob[0];
            }
        }
    }
    return parseInt(answer/numberOfJobs);
}
console.log("final",solution(jobs));

function solution(answers) {
    var answer = [];
    const answerList = [[1,2,3,4,5],[2,1,2,3,2,4,2,5],[3,3,1,1,2,2,4,4,5,5]];
    // 맞은 갯수와 id를 가지고 있는 객체를 감싸는 배열
    let rightCnt = [];
    //맞은 갯수 카운터
    let cnt=0;
    // answerList용 index
    let idx=1;
    //answerList안의 배열용 idx
    let innerIdx = 0;
    while(answerList.length > 0){
        answers.forEach((one, index)=>{
            //수포자들의 답안 배열은 shift를 이용하므로 계속 0번방을 조회화면 된다.
            one == answerList[0][innerIdx] ? cnt++ : cnt+0;
            innerIdx++;
            if(innerIdx === answerList[0].length) innerIdx = 0;
        })
        rightCnt.push({
            idx: idx,
            cnt: cnt
        });
        idx++;
        cnt=0;
        innerIdx=0;
        answerList.shift();
    }
    const max = Math.max(...rightCnt.map(one=>one.cnt))
    answer = rightCnt.filter(one=>one.cnt === max).sort((a,b)=>b.cnt-a.cnt).map(one=>one.idx)
    return answer;
}

설명#

점심시간에 도둑이 들어, 일부 학생이 체육복을 도난당했습니다. 다행히 여벌 체육복이 있는 학생이 이들에게 체육복을 빌려주려 합니다. 학생들의 번호는 체격 순으로 매겨져 있어, 바로 앞번호의 학생이나 바로 뒷번호의 학생에게만 체육복을 빌려줄 수 있습니다. 예를 들어, 4번 학생은 3번 학생이나 5번 학생에게만 체육복을 빌려줄 수 있습니다. 체육복이 없으면 수업을 들을 수 없기 때문에 체육복을 적절히 빌려 최대한 많은 학생이 체육수업을 들어야 합니다.

전체 학생의 수 n, 체육복을 도난당한 학생들의 번호가 담긴 배열 lost, 여벌의 체육복을 가져온 학생들의 번호가 담긴 배열 reserve가 매개변수로 주어질 때, 체육수업을 들을 수 있는 학생의 최댓값을 return 하도록 solution 함수를 작성해주세요.

제한사항 전체 학생의 수는 2명 이상 30명 이하입니다. 체육복을 도난당한 학생의 수는 1명 이상 n명 이하이고 중복되는 번호는 없습니다. 여벌의 체육복을 가져온 학생의 수는 1명 이상 n명 이하이고 중복되는 번호는 없습니다. 여벌 체육복이 있는 학생만 다른 학생에게 체육복을 빌려줄 수 있습니다. 여벌 체육복을 가져온 학생이 체육복을 도난당했을 수 있습니다. 이때 이 학생은 체육복을 하나만 도난당했다고 가정하며, 남은 체육복이 하나이기에 다른 학생에게는 체육복을 빌려줄 수 없습니다. 입출력 예 n lost reserve return 5 [2, 4] [1, 3, 5] 5 5 [2, 4] [3] 4 3 [3] [1] 2 입출력 예 설명 예제 #1 1번 학생이 2번 학생에게 체육복을 빌려주고, 3번 학생이나 5번 학생이 4번 학생에게 체육복을 빌려주면 학생 5명이 체육수업을 들을 수 있습니다.

function solution(numbers, target) {
    var answer = 0;
    let sum = 0;

    const dfs = (numbers, target, node) => {
        if (node === numbers.length) {
            sum = 0;
            for (let num of numbers) {
                sum += num;
            }
            if (sum === target) {
               return answer++;
            }
        } else {
            //numbers[node] *= 1
            dfs(numbers, target, node+1)

            numbers[node] *= -1
            dfs(numbers, target, node+1)
        }
    }
    dfs(numbers, target, 0);
    return answer;
}

function solution(numbers, target) {
    let answer = 0;

    function dfs(depth, sum){
        // depth가 끝에 도달했을 때 sum이 target과 같으면 answer 값 하나 올려주기
        if(depth === numbers.length){
            if(sum === target){
                answer ++;
            }
            return;
        }
        // 다음 depth로 내려가면서 뒤에 추가하는 숫자의 두가지 경우를 부르는 코드
        //                   root(0,0) [처음 0은 depth, 두번째는 sum]
        //                  /   \
        //                 /     \
        //               1,1     1,-1
        // 하나의 노드가 계속해서 dfs를 두번 부른다. 한번은 + 값과 한번은 -값으로
        // 그러다가 depth가 끝까지 들어가면 (numbers의 길이까지) sum을 target하고 비교해서 확인 가능한 것이다.
            dfs(depth+1, sum + numbers[depth])
            dfs(depth+1, sum - numbers[depth])

    }
    dfs(0,0)
    return answer;
}

function solution(n, times){
    //최소값을 1로 잡고
    let min = 1;
    times.sort((a,b)=>a-b);
    //최대값은 times에서 가장 큰 값과 n을 곱해서 구한다
    let max = n * (times[times.length-1]);
    let answer = max;


    while(min <= max){
        //이분 탐색을 하기 위해서 가운데 값을구하고
        let mid = parseInt((min + max) / 2);
        let count = 0;

        //times에 mid값에 몇명이 처리 될 수 있는지를 count에 누적
        times.forEach(one=>{
            count += parseInt(mid / one);
        });

        //누적된 count값이 n을 넘거나 같으면 일단 answer에 mid나 answer중에서 작은 값을 대입
        if(count >= n){
            answer = Math.min(answer, mid);
        }

        // count가 크다는 말은 시간을 줄여도 될다는 말이니 max값을 mid에서 1 낮추고 다시 while돌기
        if(count >= n){
            max = mid - 1;
        // count가 작다는 이야기는 처리해야하는 사람수를 처리 못한 것이니 min을 mid에서 1 늘리고 다시 while돌기
        }else{
            min = mid + 1;
        }
    }
    return answer;
}

    const queue = [1]; // bfs를 하기위한 노드=> 여기서 1번을 줘서 queue돌 때 1부터 돈다
    const adjEdgeList = new Array(n + 1).fill(null).map(() => Array()); // 간선 배열
