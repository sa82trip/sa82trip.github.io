how to call function!#

  1. normal way

         function shout(phrase){
     console.log(phrase)
     }
     shout("I Love You!!!")

         #+RESULTS:
     : I Love You!!!
     : undefined

  2. call func as member function

         const tarzan = {
         shout: function (phrase) {
             console.log(phrase);
         }
     }
     tarzan.shout("Yahooooooooooooooo!")

         #+RESULTS:
     : Yahooooooooooooooo!
     : undefined

  3. call and apply from Function.prototype

         console.log(Function.prototype.call)
     console.log("======================")
     console.log(Function.prototype.apply)

         #+RESULTS:
         : [Function: call]
         : ======================
         : [Function: apply]
         : undefined

         function shout(phrase){
         console.log(phrase);
     }
     shout.call(undefined, "Hey!!!!!");
     shout.apply(undefined, ["Hey!!!!!"])

         #+RESULTS:
     : Hey!!!!!
     : Hey!!!!!
     : undefined

    call, apply는 첫번 째 인자로 자신을 부르는 주체를 받는다.

function tellsThis(){
    console.log(this.toString());
}

const objectWithTeller = {
    tell: tellsThis,
    toString: function () {
        return "[object testObject]"
    }
        }

//2. call the one in the object
objectWithTeller.tell();

tellsThis.apply(objectWithTeller);
//5.
objectWithTeller.tell.call(undefined);
//6.
objectWithTeller.tell.apply(objectWithTeller);

//7.
copyOfTell();

this가 binding되는 법칙#

  1. strict mode면 thisArg가 this
  2. thisArg가 null or undefined면 this는 global object
  3. thisArg가 Object가 아니면 thisArg를 Object화
  4. 위가 다 아니면 thisArg는 this
function bindingTest () {
    return this;
}
console.log(bindingTest())
console.log(bindingTest.call(true))
console.log(bindingTest.call("hello"))
console.log(bindingTest.call(3))
console.log(bindingTest.call(['test']))

Array의 slice 함수 이용하기#

arguments가 array like object이지 array는 아니기 때문에 slice같은 array가 가지고 있는 함수는 사용할 수 없습니다만 위에서 봤던 call이나 apply를 이용하면 가능합니다.

function argTest(){
    return arguments;
}
const args = argTest(1,2,3);
console.log(args)
console.log(args.slice(1))

#+RESULTS:
console.log(args.slice(1))}()));
                 ^

TypeError: args.slice is not a function

slice함수를 가지고 있지 않기 때문에 위와 같은 에러가 발생합니다.

call or apply를이용하여 Array의 slice 함수 이용하기

function argTest(){
    return arguments;
}
const args = argTest(1,2,3);
const slicedArgs = Array.prototype.slice.call(args,0);
const slicedArgs1 = Array.prototype.slice.call(args,1);
const slicedArgs2 = Array.prototype.slice.apply(args,[2]);
console.log(slicedArgs);
console.log(slicedArgs1);
console.log(slicedArgs2);

#+RESULTS:
: [ 1, 2, 3 ]
: [ 2, 3 ]
: [ 3 ]

var myObject = function () {
    var value = 0;

    return {
        increment: function (inc) {
            value += typeof inc === 'number' ? inc : 1;
        },
        getValue: function () {
            return value;
        }
    }
}
var object_with_closure = myObject();
console.log(object_with_closure.getValue());
console.log(object_with_closure.increment(5));
console.log(object_with_closure.getValue());

13번째 줄에서 var value에 접근이 가능한 함수를 반환하고 있습니다. value값을 가져오고 싶을 땐 getValue()를 이용해서 가져올 수 있고, 값을 더하려면 increment를 이용하면 됩니다. 이렇게 구성하므로써 closure를 이용하여 value를 안전(직접 접근하는 것이 아닌 setter, getter를 이용하기 때문에)하게 사용 할 수 있게 되었습니다.

const objectA = {
    name: "local variable",
    say: function (e) {
        console.log("Hello " + this.name);
    }
};

objectA.say();
const speak = objectA.say;
speak();
//임시함수를 이용해서 binding
const bindedSpeak = function (e) { objectA.say(); };
bindedSpeak();

//closure의 특성을 이용해서 binding용 함수 만들기
function bind(obj, func){
    return function(){
        func.apply(obj, arguments);
    };
}
const contextTestObj ={
    name:"name for binded test"
}
// 만들어본 bind함수 이용하기
bind(contextTestObj, speak)();
//es5부터 탑재된 bind 함수 이용하기
speak.bind(contextTestObj)();

#+RESULTS:
: Hello local variable
: Hello undefined
: Hello local variable
: Hello name for binded test
: Hello name for binded test
: undefined
